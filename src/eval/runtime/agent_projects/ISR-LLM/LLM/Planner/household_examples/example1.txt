(define (domain household)
(:types object location container)
(:predicates
(at-robot ?l - location)
(hand-empty)
(holding ?o - object)
(at ?o - object ?l - location)
(is-open ?c - container)
(is-closed ?c - container)
(is-cooked ?o - object)
(is-sliced ?o - object)
)
(:action navigate
  :parameters (?from - location ?to - location)
  :precondition (and (at-robot ?from))
  :effect (and (at-robot ?to) (not (at-robot ?from)))
)
(:action pick-up
  :parameters (?o - object ?l - location)
  :precondition (and (at-robot ?l) (at ?o ?l) (hand-empty))
  :effect (and (holding ?o) (not (at ?o ?l)) (not (hand-empty)))
)
(:action place
  :parameters (?o - object ?l - location)
  :precondition (and (at-robot ?l) (holding ?o))
  :effect (and (at ?o ?l) (not (holding ?o)) (hand-empty))
)
(:action open
  :parameters (?c - container ?l - location)
  :precondition (and (at-robot ?l) (at ?c ?l) (is-closed ?c))
  :effect (and (is-open ?c) (not (is-closed ?c)))
)
(:action close
  :parameters (?c - container ?l - location)
  :precondition (and (at-robot ?l) (at ?c ?l) (is-open ?c))
  :effect (and (is-closed ?c) (not (is-open ?c)))
)
(:action cook
  :parameters (?o - object ?l - location)
  :precondition (and (at-robot ?l) (holding ?o))
  :effect (and (is-cooked ?o))
)
(:action slice
  :parameters (?o - object ?l - location)
  :precondition (and (at-robot ?l) (holding ?o))
  :effect (and (is-sliced ?o))
)

(define (problem slice_apple)
(:domain household)
(:objects apple knife - object
kitchen_counter dining_table - location)
(:init
(at-robot kitchen_counter)
(hand-empty)
(at knife kitchen_counter)
(at apple dining_table)
)
(:goal
(and
(is-sliced apple)
(at apple dining_table))
)

Action Sequence:
Step 1: pick up the knife at the kitchen_counter
(pick-up knife kitchen_counter)
Step 2: navigate to the dining_table where the apple is, place the knife so the hand is free, then pick up the apple
(navigate kitchen_counter dining_table)
(place knife dining_table)
(pick-up apple dining_table)
Step 3: slice the apple while holding it at the dining_table, then place it back
(slice apple dining_table)
(place apple dining_table)
The apple is sliced and remains on the dining_table.
